public class EmployeeServiceTests
{
	private readonly Mock<IUnitOfWork> _uow;
	priavet readonly Mock<IGenericRepository<Employee>> _repoMock;
	private readonly EmployeeService _service;
	
	public EmployeeServiceTests()
	{
		_uow = new Mock<IUnitOFWork>();	
		_repo = new Mock<IGenericRepository<Employee>>();

		_uow.Setup(x => x.Employees).Returns(_repo.Object);
		
		_service = new EmployeeService(_uow.Object);
	}
	
	[Fact]
	public async Task CreateEmployeeAsync_ShouldCreateEmployee()
	{
		var dto = new CreateEmployeeDto("Sujal", 50000, 1);
			
		Employee? addedEmployee = null;
	
		_repo.Setup(x => x.AddAsync(It.IsAny<Employee>()))
			.Callback<Employee>(emp => {addedEmployee = emp})
			.Returns(Task.CompletedTask);

		var result = await _service.CreateEmployeeAsync(dto);

		Assert.NotNull(result);

		Assert.NotNull(addedEmployee);
		
		Assert.Equal(result.Name, addedEmployee.Name);
		Assert.Equal(result.Email, addedEmployee.Email);

		_repo.Verify(x => x.AddAsync(It.IsAny<Employee>()),
			Times.Once);
		
	}
}
