Develop a Library Management System where members can borrow books, librarians manage inventory, and the system tracks borrowing history, penalties, and reports.
Requirements
A Book can have multiple copies.
At a time, one copy can only be issued to one member.
A Member can borrow a maximum of 5 books.

Book issue period is 15 days.
If returned late, charge ₹10 per day.

If all copies are unavailable, add the member to a waiting list.
Notify the next member when a book is returned.
Store complete borrowing history.

Copy
Member
BorrowHistory

public abstract class BaseEntity
{
	public Guid Id {get; protected set;} = Guid.NewGuid();
}

public class Book : BaseEntity
{
	public string Title {get; set;}

	public int TotalCount {get; set;}
	public int AvailableCount {get; set;}
	
	public void IncreaseCopy(int count)
	{
		TotalCount += count;
		AvailableCount += count;
	}
	public void DecreaseCopy(int count)
	{
		if(AvailableCount < count)
			throw new InvalidOperationException("Book copy is not available");
		TotalCount -= count;
		AvailableCount -= count;

	}
	
	public ICollection<BookCopy> BookCopies {get; set;} = new List<BookCopy>();
	public ICollection<WaitingListEntry> WaitingListEntry {get; set;} = new List<WaitingListEntry>();

}

public class BookCopy  : BaseEntity
{

	public Guid BookId {get; set;}
	public Book Book {get; set;}
	
	
	
	
	public ICollection<BorrowHistory> BorrowHistorys {get; set;} = new List<BorrowHistory>();
	
	public bool IsAvailable{get; set;} = true;
	
	public void SetAvailable()
	{
		IsAvailable = true;
	}
	public void SetUnAvailable()
	{
		IsAvailable = false;
	}
	


}

public class Member : BaseEntity
{
	public string Name{get; set;}
	public int AquiredBookCount{get; set;}
	
	public void IncreaseBookCount()
	{
		if(AquiredBookCount == 5)
			throw new InvlaidOperationException("Member has Reached Max book count");

		AquiredBookCount++;
	}
	public void DecreaseBookCount()
	{
		if(AquiredBookCount <= 0)
			throw new InvlaidOperationException("Member has not any book currently");

		AquiredBookCount--;
	}


	public ICollection<BookCopy> BookCopies {get; set;} = new List<BookCopy>();
	
	public ICollection<BorrowHistory> BorrowHistorys {get; set;} = new List<BorrowHistory>();
	public ICollection<WaitingListEntry> WaitingListEntry {get; set;} = new List<WaitingListEntry>();
	
}

public class BorrowHistory : BaseEntity
{
	public Guid MemberId{get; set;}
	public Member Member {get; set;}
	
	public Guid BookCopyId{get; set;}
	public BookCopy BookCopy {get; set;}
	
	public DateTime BorrowedAt{get; set;}
	public DateTime? ReturnedAt{get; set;}
	public Datetime DueAt{get; set;}
	
	public decimal OverDueCharge{get; set;}

	public BorrowHistory(Guid memberId, Guid bookcopyId, DateTime borrowedAt)
	{
		MemberId = memberId;
		BookCopyId = bookcopyId;
		BorrowedAt = borrowedAt;
		DueAt = borrowedAt.AddDays(15);
	}
	public void Return(DateTime returnDate)
	{
		var OverDueDays = returnDate - DueAt;
		
		if(OverDueDays > 0)
		{
			CalculateOverdueCharge(OverDueDays);
		}
		
		ReturnedAt = returnDate;
		
	}
	
	private void CalculateOverdueCharge(int overDueDays)
	{
		OverDueCharge = overDueDays * 10m;
	}
}

public class WaitingListEntry : BaseEntity
{
	
	public WaitingStatus Status {get; set;} = WaitingStatus.Waiting;

	public Guid MemberId{get; set;}
	public Member Member {get; set;}
	
	public Guid BookId{get; set;}
	public Book Book{get; set;}
	
	public DateTime RequestedAt{get; set;} = DateTime.Now;
	public DateTime? NotifiedAt{get; set;}
	public DateTime? CompletedAt{get; set;}
	
	public void Notify(DateTime notify)
	{
		NotifiedAt = notify;
		Status = WaitingStatus.Notified;	
	}
	public void Complete(DateTime completedAt)
	{
		CompletedAt = completedAt;
		Status = WaitingStatus.Completed;
	}


}
public Enum WaitingStatus
{
	Waiting,
	Notified,
	Completed
}

public class AppDbContext : DbContext
{
	
	public AppDbContext(DbContextOptions<AppDbcontext> options) : base(options)
	{}
	
	public DbSet<Book> Books => DbSet<Book>();
	public DbSet<Member> Members => DbSet<Member>();
	public DbSet<BookCopy> BookCopies => DbSet<BookCopy>();
	public DbSet<BorrowHistory> Books => DbSet<BorrowHistory>();
	
	public DbSet<WaitingListEntry> WaitingLists => DbSet<WaitingListEntry>();

	protected override void OnModelCreating(ModelBuilder builder)
	{
		base.OnModelCreating(builder);

		ConfigureBorrowHistory(builder);
	}

	public override Task<int> SaveChangesAsync()
	{
		base.SaveChangesAsync();
	}
	
	private static void ConfigureBorrowHistory(ModelBuilder builder)
	{
		builder.Entity<BorrowHistory>(e => {
		
			e.ToTable("BorrowHistories");
			e.HasKey(x => x.Id);
			
			e.HasOne(x => x.Member)
			.WithMany(x => x.BorrowHistorys)
			.HasForeignKey(x => x.MemberId)
			.OnDelete(DeleteBehaviour.Restrict);
	
			e.HasOne(x => x.BookCopy)
			.WithMany(x => x.BorrowHistorys)
			.HasForeignKey(x => x.BookCopyId)
			.OnDelete(DeleteBehaviour.Restrict);

		});
	}
	

}

public interface IGenericRepository<T> where T : BaseEntity
{
	Task<T?> GetByIdAsync(Guid id);
	Task<IEnumerable<T>> GetAllAsync();
	Task AddAsync(T entity);
	void Update(T entity);
	void Delete(T entity);

}
public class GenericRepository<T>(AppDbContext context) : IGenericRepository<T> where T : BaseEntity
{
	private readonly DbSet _set = context.Set<T>();
	
	public async Task<T?> GetByIdAsync(Guid id)
	{
		return await _set.FindAsync(id);
	}
	public async Task<IEnumerable<T>> GetAllAsync()
	{
		return await _set.ToListAsync();
	}
	public async Task AddAsync(T entity)
	{
		await _set.AddAsync(entity);
	}
	public void Update(T entity)
	{
		_set.Update(entity);
	}
	
	public void Delete(T entity)
	{
		_set.Remove(entity);
	}
}

	public IGenericRepository<WaitingListEntry> WaitingListEntrys{get;} = new GenericRepository<WaitingListEntry>(context);
	public IGenericRepository<BorrowHistory> BorrowHistorys{get;} = new GenericRepository<BorrowHistory>(context);
	
	public async Task<int> SaveChangesAsync()	
	{
		return await context.SavechangesAsync();
	}
}
