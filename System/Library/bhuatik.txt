public abstract class BaseEntity
{
	public Guid Id {get; protected set;} = Guid.NewGuid();
}

public class Member(string name, string email) : BaseEntity
{
	public string Name {get; set;} = name;
	public string Email{get; set;} = email;
	
	public ICollection<BorrowRecord> BorrowRecords {get; set;} = new List<BorrowRecord>();
	
}

public class Book : BaseEntity
{
	public string Title {get; set;}
	public string ISBN {get; set;}

	public int TotalCopy {get; set;}
	public int AvailableCopy {get; set;}

	public ICollection<BookCopy> BookCopies {get; set;} = new List<BookCopy>();

	public void AddCopy(string copyNumber)
	{
		var copy = new BookCopy(Id, copyNumber);
		BookCopies.Add(copy);
		TotalCopy++;
		AvailableCopy++;	
	}
	public void IncreaseAvailableCopy()
	{
		AvailableCopy++;
	}
	public void DecreaseAvailableCopy()
	{
		if(AvailableCopy <= 0)
			throw new NotAvailableException("No books available");
		AvailableCopy--;
	}

	
	
}
public class BookCopy : BaseEntity
{
	public string CopyNumber {get; set;}
	
	public Guid BookId {get; set;}
	public Book Book {get; set;}
	
	public bool IsAvailable {get; set;} = true;

	public ICollection<BorrowRecord> BorrowRecords {get; set;} = new List<BorrowRecord>();
	
	public BookCopy(Guid id, string number)
	{
		BookId = id;
		CopyNumber = number;
		IsAvailable = true;
	}
	
	public void MarkBorrowed()
	{
		if(!IsAvailable)
			throw new NotAvailableException("book is not available");
		Is Available = false;
	}
	public void MarkReturned()
	{
		IsAvailable = true;
	}

}

public static class LibraryPolicy
{
	public const int BorrowPeriodDays = 10;
	public const int MaxOverDueAllowed = 5;	
}

public class BorrowRecord : BaseEntity
{
	public DateTime BorrowedAt {get; set;}
	public DateTime? RetunedAt  {get; set;}
	public DateTime DueAt  {get; set;}
	
	public Guid MemberId {get; set;}
	public Member Member {get; set;} = null!;

	public Guid BookCopyId {get; set;}
	public BookCopy BookCopy {get; set;}  = null!;
	
	public bool IsReturned{get; private set;} = false;
	
	public bool IsOverDue{get; private set;} = false;

	
	public BorrowRecord(Guid memberId, Guid bookCopyId, DateTime borrowedAt)
	{
		MemberId = memberId;
		BookCopyId = bookCopyId;
		BorrowedAt = borrowedAt;
		
		DueAt = borrowedAt.AddDays(LibraryPolicy.BorrowPeriodDays);
	}
	
	public void Return(DateTime returnedAt)
	{
		
		if(returnedAt > DueAt)
			IsOverDue = true;
		
		ReturnedAt = returnedAt;
	}
}
